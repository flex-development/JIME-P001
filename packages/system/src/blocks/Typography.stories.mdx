## Variables

File: `packages/system/src/scss/variables/_settings.scss`

```scss
$font-family-monospace: SFMono-Regular, Menlo, 'Courier New', monospace;
$font-family-sans-serif: ibm-plex-sans, sans-serif;
$font-family-sans-serif-special: quimby-gubernatorial, sans-serif;

$font-family-base: var(--#{$variable-prefix}font-sans);
$font-family-code: var(--#{$variable-prefix}font-mono);
$font-family-special: var(--#{$variable-prefix}font-sans-special);

// $font-size-root effects the value of `rem`
$font-size-root: null;

// $font-size-base effects the font size of the body text
$font-size-base: rem(16);

$font-size-xs: rem(12); // 12px
$font-size-sm: rem(14); // 14px
$font-size-md: $font-size-base; // 16px
$font-size-lg: rem(18); // 18px
$font-size-xl: rem(20); // 20px
$font-size-xxl: rem(24); // 24px

$font-weight-hairline: 100;
$font-weight-thin: 200;
$font-weight-light: 300;
$font-weight-normal: 400;
$font-weight-medium: 500;
$font-weight-semibold: 600;
$font-weight-bold: 700;
$font-weight-extrabold: 800;
$font-weight-black: 900;

$font-weight-base: $font-weight-medium;

$letter-spacing: letter-spacing(-48);

$line-height-base: strip-unit($space-24);
$line-height-sm: strip-unit($space-20);
$line-height-lg: strip-unit($space-32);

$h1-font-size: $space-32;
$h2-font-size: $space-28;
$h3-font-size: $font-size-xxl;
$h4-font-size: $font-size-xl;
$h5-font-size: $font-size-lg;
$h6-font-size: $font-size-base;

$headings-margin-bottom: $spacer / 2;
$headings-margin-top: $space-0;
$headings-font-family: null;
$headings-font-style: null;
$headings-font-weight: $font-weight-bold;
$headings-line-height: $line-height-base;
$headings-color: $secondary;
$headings-text-transform: uppercase;
```
